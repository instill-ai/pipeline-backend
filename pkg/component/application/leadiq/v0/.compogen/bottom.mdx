## Example Recipes


```yaml
version: v1beta

variable:
  city:
    title: City
    type: string
    default: "London"
  country:
    title: Country
    type: string
    default: "United Kingdom"
  industry:
    title: Industry
    type: string
    default: "Software Development"
  description:
    title: Description
    type: string
    default: "AI"
  technology:
    title: Technology
    type: string
    default: "HTTP"

component:
  prospects:
    type: leadiq
    task: TASK_FIND_PROSPECTS
    input:
      company:
        names:
        revenue-size:
          min: 10000000
          max:
        countries:
          - ${variable.country}
        states:
        cities:
          - ${variable.city}
        industries:
          - ${variable.industry}
        descriptions:
          - ${variable.description}
        technologies:
          - ${variable.technology}
      limit: 2
      filter-by:
        job-title: (?i)\b(sales|founder)\b
        seniorities:
          - Executive
          - Director
        function:
    condition:

output:
  result:
    title: Result
    value: ${prospects.output}
```
